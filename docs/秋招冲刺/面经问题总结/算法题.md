### 1.求根号2

```c++
double MySqrt(double n)
{
    double _max = n;
    double _min = 0.0;
    // 精度
    double p = 1e-5;
    double mid = (_max + _min)/2.0;
    // 浮点数差的绝对值与精度比较
    while(fabs(mid*mid) > n)
    {
    	if(mid*mid > n)
        {
            _max = mid;
        }
        else if(mid*mid < n)
        {
            _min = mid;
        }
        else
        	return mid;
        mid = (_max + _min)/2.0;
	}
    return mid;
}
```
